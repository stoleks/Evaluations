  \variationSujet{
    L'ammoniac est un gaz irritant à température ambiante.

  }{

  }

  \begin{center}
    \variationSujet{

      }

      \chemfig[atom sep=30pt, atom style={scale=1.5}, line width=6pt]{
        H - \charge{-90:4pt=\|}{P} (-[3] H) - H
      }
    }

  \end{center}
